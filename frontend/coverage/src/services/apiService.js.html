/**
 * Simulates an API call with realistic typing effect
 * @param {string} message - The user's message
 * @param {function} onChunk - Callback function that receives response chunks
 * @returns {Promise<void>}
 */
export const mockApiCall = async (message, onChunk) => {
  await new Promise(resolve => setTimeout(resolve, 50)) // Initial thinking delay
 
  // More realistic response with multiple sentences
  const fullResponse =
    "Thank you for your message. I'm an AI assistant designed to help answer your questions and engage in conversation. I can provide information on a wide range of topics, assist with tasks, or just chat. What else would you like to know about today?"
 
  // Split the response into words for more natural streaming
  const words = fullResponse.split(' ')
  let currentResponse = ''
 
  for (let i = 0; i < words.length; i++) {
    // Add the next word with a space (except for the first word)
    currentResponse += (i === 0 ? '' : ' ') + words[i]
 
    // Simulate variable typing speed for natural effect
    const delay = Math.floor(Math.random() * 40) + 10 // Random delay between 10-50ms
 
    // Add occasional "thinking" pauses at punctuation
    const hasPunctuation = /[.!?]/.test(words[i])
    const thinkingTime = hasPunctuation ? Math.random() * 300 : 0
 
    await new Promise(resolve => setTimeout(resolve, delay + thinkingTime))
    onChunk(currentResponse)
  }
}
